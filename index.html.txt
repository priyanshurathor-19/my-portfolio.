    <script>
        function toggleMenu() {
            const nav = document.getElementById('navLinks');
            nav.classList.toggle('active');
        }

        // CHATBOT LOGIC
        const chatWindow = document.getElementById('chatWindow');
        const chatMessages = document.getElementById('chatMessages');
        const chatInput = document.getElementById('chatInput');
        const apiKey = ""; // API Key injected at runtime

        function toggleChat() {
            chatWindow.style.display = chatWindow.style.display === 'flex' ? 'none' : 'flex';
        }

        function handleEnter(e) {
            if (e.key === 'Enter') sendMessage();
        }

        function appendMessage(text, sender) {
            const div = document.createElement('div');
            div.classList.add('message', sender === 'user' ? 'user-msg' : 'bot-msg');
            div.textContent = text;
            chatMessages.appendChild(div);
            chatMessages.scrollTop = chatMessages.scrollHeight;
        }

        async function sendMessage() {
            const text = chatInput.value.trim();
            if (!text) return;

            appendMessage(text, 'user');
            chatInput.value = '';
            
            // Loading indicator
            const loadingDiv = document.createElement('div');
            loadingDiv.classList.add('message', 'bot-msg', 'loading-dots');
            loadingDiv.textContent = "Thinking";
            chatMessages.appendChild(loadingDiv);
            chatMessages.scrollTop = chatMessages.scrollHeight;

            try {
                // System prompt with context about Priyanshu
                const systemPrompt = `
                    You are a friendly and helpful AI assistant for Priyanshu Rathore's personal portfolio website.
                    Here is the context about Priyanshu:
                    - Name: Priyanshu Rathore
                    - Role: 12th Grade Student, Beginner Web Developer, Photographer, Videographer.
                    - Location: Kota, Rajasthan, India.
                    - Skills: HTML, CSS, Basic JavaScript, UI/UX Design interest, Photography, Videography.
                    - Contact Email: priyanshurathor612@gmail.com
                    - Contact Phone: +91 74130 78090
                    - Instagram: _pr1yanshu_
                    - Goal: To showcase his beginner journey and find opportunities to learn.
                    
                    Instructions:
                    - Answer questions as if you are representing him.
                    - Keep answers concise, professional, yet friendly (he is a student).
                    - If asked about complex coding topics, admit he is a beginner but eager to learn.
                    - If asked for "Source Code", say this site is built with HTML/CSS.
                `;

                const response = await fetchWithRetry(text, systemPrompt);
                
                chatMessages.removeChild(loadingDiv);
                appendMessage(response, 'bot');

            } catch (error) {
                console.error(error);
                chatMessages.removeChild(loadingDiv);
                appendMessage("Oops! I'm having trouble connecting to my brain right now. Try again later.", 'bot');
            }
        }

        async function fetchWithRetry(userText, systemPrompt, retries = 3, delay = 1000) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: userText }] }],
                systemInstruction: { parts: [{ text: systemPrompt }] }
            };

            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });

                    if (!response.ok) throw new Error(`HTTP error! status: ${response.status}`);
                    
                    const data = await response.json();
                    return data.candidates[0].content.parts[0].text;
                } catch (err) {
                    if (i === retries - 1) throw err;
                    await new Promise(res => setTimeout(res, delay * Math.pow(2, i)));
                }
            }
        }
